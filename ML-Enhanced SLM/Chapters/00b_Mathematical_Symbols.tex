\chapter*{Mathematical Symbols}
\addcontentsline{toc}{chapter}{Mathematical Symbols}
\label{ch:mathematical_symbols}

\begin{center}
\begin{tabular}{l l}
\toprule
    \textbf{Symbol} & \textbf{Definition} \\
\midrule
$N$ & Number of OFDM subcarriers \\
$k$ & Subcarrier index ($k = 0, 1, \dots, N-1$) \\
$n$ & Time-domain sample index ($n = 0, 1, \dots, N-1$) \\
$U$ & Number of SLM candidate phase sequences \\
$u$ & Index of a specific SLM candidate ($u = 1, \dots, U$) \\
$\hat{u}$ & Index of the selected (best) SLM candidate \\
$M$ & Phase alphabet size \\
$V$ & Number of PTS sub-blocks \\
$L$ & Number of unfolded layers (PR-DUN) or oversampling factor \\
$\bm{X}$ & Frequency-domain OFDM data symbol vector \\
$X_k$ & Complex symbol on the $k$-th subcarrier \\
$x(n)$ & Time-domain OFDM signal sample at index $n$ \\
$\bm{x}$ & Time-domain OFDM signal vector \\
$\bm{s}$ & Transmitted signal (after PAPR reduction) \\
$\bm{r}$ & Received signal \\
$\bm{c}$ & Peak-cancelling signal (NN-SCF, PR-DUN) \\
$\bm{n}$ & AWGN noise vector \\
$\bm{\psi}_u$ & $u$-th SLM phase sequence vector \\
$\psi_{u,k}$ & Phase factor for subcarrier $k$ in sequence $u$ \\
$\varphi_{u,k}$ & Phase angle of $\psi_{u,k}$ \\
$\bm{W}$ & Weight matrix of a neural network layer \\
$\bm{b}$ & Bias vector of a neural network layer \\
$f(\cdot)$ & Activation function \\
$\mathcal{L}$ & Loss function \\
$d_x$ & Input dimension (autoencoder) \\
$d_h$ & Latent representation dimension (autoencoder) \\
$A$ & Clipping threshold \\
\bottomrule
\end{tabular}
