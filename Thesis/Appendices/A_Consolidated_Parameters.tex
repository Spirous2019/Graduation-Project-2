\chapter{PRNet Simulation Parameters}
\label{app:consolidated_params}

This appendix consolidates all simulation and system parameters used in
the PRNet experiments, as reported in Kim et al.\ (2017)~\cite{kim2017novel}.

\begin{table}[H]
    \centering
    \caption{System and training parameters for PRNet~\cite{kim2017novel}.}
    \label{tab:prnet_full_params}
    \begin{tabular}{@{} l l @{}}
        \toprule
        \textbf{Parameter} & \textbf{Value} \\
        \midrule
        \multicolumn{2}{@{}l}{\textit{OFDM System}} \\
        Number of subcarriers ($N$)             & 64 \\
        Modulation                              & QPSK \\
        Oversampling factor ($L$)               & 4 \\
        Channel (training)                      & AWGN / Rayleigh flat fading \\
        \midrule
        \multicolumn{2}{@{}l}{\textit{Network Architecture}} \\
        Hidden layers per DNN (encoder/decoder) & 5 \\
        Neurons per hidden layer                & 2048 \\
        Hidden activation                       & ReLU \\
        Output activation                       & Linear \\
        Batch Normalization                     & Yes (after every FC layer) \\
        \midrule
        \multicolumn{2}{@{}l}{\textit{Training Configuration}} \\
        Optimizer                               & Adam \\
        Learning rate                           & $10^{-4}$ \\
        Batch size                              & 400 \\
        Training steps                          & 500{,}000 \\
        Optimal corruption level ($\eta^\star$) & $-15$~dB \\
        PAPR loss weight ($\lambda$)            & 0.01 \\
        \midrule
        \multicolumn{2}{@{}l}{\textit{Reported Results}} \\
        PAPR reduction at CCDF $= 10^{-3}$      & $\approx 6$--$7$~dB \\
        Inference speedup vs.\ PTS (GPU)        & ${\approx}6\times$ \\
        \bottomrule
    \end{tabular}
\end{table}

\begin{keyinsight}
    The two key hyperparameters that define PRNet's operating point are:
    \begin{itemize}
        \item $\eta^\star = -15$~dB --- the optimal training corruption level,
              identified in Stage~1 by sweeping
              $\eta \in \{0, -5, -10, -15, -20, -25, -30\}$~dB.
        \item $\lambda = 0.01$ --- the PAPR loss weight, which makes
              reconstruction 100$\times$ more important than PAPR reduction
              in the joint loss, preserving BER while still achieving
              significant PAPR gains.
    \end{itemize}
\end{keyinsight}
