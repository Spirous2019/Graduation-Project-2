% Redefine \cleardoublepage so that blank verso pages inserted by
% twoside/openright are completely empty (no stray page number).
\makeatletter
\renewcommand{\cleardoublepage}{%
    \clearpage
    \if@twoside
        \ifodd\c@page\else
            \hbox{}\thispagestyle{empty}\newpage
            \if@twocolumn\hbox{}\newpage\fi
        \fi
    \fi
}
\makeatother

% THESIS style:
%   Even (left)  pages : chapter name on LEFT  (outer edge)
%                        page number  on RIGHT (inner edge)
%   Odd  (right) pages : page number  on LEFT  (inner edge)
%                        chapter name on RIGHT (outer edge)
\fancypagestyle{thesis}{%
    \fancyhf{}
    \fancyhead[LE]{\textit{\nouppercase{\leftmark}}}   % Chapter — outer left  (even)
    \fancyhead[RO]{\textit{\nouppercase{\leftmark}}}   % Chapter — outer right (odd)
    \fancyhead[RE]{\thepage}                            % Page #  — inner right (even)
    \fancyhead[LO]{\thepage}                            % Page #  — inner left  (odd)
    \renewcommand{\headrulewidth}{0.4pt}
    \renewcommand{\headrule}{%
        \hbox to\headwidth{%
            \color{asuBlue}\leaders\hrule height \headrulewidth\hfill}}
    \renewcommand{\footrulewidth}{0pt}
}
